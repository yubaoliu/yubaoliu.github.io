3 Python Function

3.1 se2

import math
import numpy as np
def se2(a, b, c):
        cth = math.cos(c)
        sth = math.sin(c)
        R = np.array([[cth, -sth, a],
                                  [sth, cth, b],
                                  [0, 0, 1]])
        return R

print(se2(1, 2, math.pi/6))
